Methods

Method
read([options])
Type: object

This always return MP3Tag.tags

            
              // Default Options
              {
                id3v1: true, // do not ignore ID3v1
                id3v2: true // do not ignore ID3v2
              }
Method
save([options])
Type: ArrayBuffer|Buffer

This always return MP3Tag.buffer

            
              // Default Options
              {
                strict: false, // Use strict mode when validating
                id3v1: { include: true }, // Add ID3v1 tags to buffer
                id3v2: {
                  include: true, // Add ID3v2 tags to buffer
                  unsynch: true, // Use unsynchronisation scheme
                  version: defaultVersion, // ID3v2 major version
                  padding: 2048 // Number of null bytes added after ID3v2
                }
              }

getAudio()
